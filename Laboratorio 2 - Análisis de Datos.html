In¬†[23]:
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

mejor_mse = float('inf')  
mejor_modelo = None       
mejores_parametros = {}   

for a in parametros["alphas"]:
    for r in parametros["ratio"]:
        elastic_net = ElasticNet(alpha=a, l1_ratio=r, random_state = 42)
        elastic_net.fit(X_train, y_train)

        # Predecir en el conjunto de prueba
        y_pred = elastic_net.predict(X_test)
        
        # Calcular el MSE
        mse = mean_squared_error(y_test, y_pred)

        # Verificar si es el mejor modelo
        if mse < mejor_mse:
            mejor_mse = mse
            mejor_modelo = elastic_net
            mejores_parametros = {"alpha": a, "l1_ratio": r}

# Resultados
print(f"Mejor MSE: {mejor_mse:.4f}")
print(f"Mejores par√°metros: {mejores_parametros}")
